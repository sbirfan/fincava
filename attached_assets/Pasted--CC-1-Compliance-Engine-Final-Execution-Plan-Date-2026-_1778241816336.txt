### Route 1 — Priority Queue

```typescript
// GET /api/admin/compliance/queue?offset=0&limit=100
// Returns suppliers ranked by compliance priority score (admin only).
// priority = 0.40×sellable_proximity + 0.30×ai_score + 0.20×blocker_score + 0.10×staleness
router.get('/compliance/queue', ...adminOnly, async (req, res) => {
  const offset = parseInt(String(req.query.offset ?? '0'), 10);
  const limit  = Math.min(parseInt(String(req.query.limit ?? '100'), 10), 200);

  const queue = await db.execute(sql`
    WITH blocker_counts AS (
      SELECT supplier_id,
             COUNT(*) FILTER (WHERE state IN (
               'not_started','not_sure','self_serve_in_progress',
               'assisted_in_progress','needs_fix','submitted'
             )) AS open_blockers
      FROM supplier_requirement_status
      GROUP BY supplier_id
    ),
    last_review AS (
      SELECT supplier_id,
             MAX(reviewed_at) AS last_reviewed_at
      FROM admin_compliance_reviews
      GROUP BY supplier_id
    ),
    ai_scores AS (
      SELECT DISTINCT ON (supplier_id) supplier_id, commercial_score
      FROM supplier_evaluations
      ORDER BY supplier_id, created_at DESC
    ),
    proximity AS (
      SELECT id,
        CASE sellable_status
          WHEN 'PUBLISHED'  THEN 1.00
          WHEN 'SELLABLE'   THEN 0.85
          WHEN 'ELIGIBLE'   THEN 0.50
          ELSE 0.10
        END AS sellable_proximity
      FROM suppliers
    )
    SELECT
      s.id,
      s.nombre_completo        AS display_name,
      s.sellable_status,
      s.supplier_type,
      COALESCE(ai.commercial_score, 0)  AS ai_quality_score,
      COALESCE(bc.open_blockers, 0)     AS open_blockers,
      lr.last_reviewed_at,
      ROUND((
        0.40 * p.sellable_proximity +
        0.30 * (COALESCE(ai.commercial_score, 0) / 100.0) +
        0.20 * GREATEST(0, 1 - COALESCE(bc.open_blockers, 0) / 10.0) +
        0.10 * LEAST(1, EXTRACT(EPOCH FROM (
          NOW() - COALESCE(lr.last_reviewed_at, s.created_at)
        )) / (30 * 86400))
      )::numeric, 3) AS priority_score
    FROM suppliers s
    LEFT JOIN proximity       p  ON p.id            = s.id
    LEFT JOIN ai_scores       ai ON ai.supplier_id  = s.id
    LEFT JOIN blocker_counts  bc ON bc.supplier_id  = s.id
    LEFT JOIN last_review     lr ON lr.supplier_id  = s.id
    ORDER BY priority_score DESC
    LIMIT ${limit} OFFSET ${offset}
  `);

  res.json({ queue: queue.rows, offset, limit });
});
```

### Route 3 — Review + State Transition

```typescript
// POST /api/admin/compliance/supplier/:supplierId/review
// Admin reviews a requirement: writes audit log and transitions state.
const reviewBodySchema = z.object({
  requirementCode: z.enum(['DIAN_RUT', 'ICA_CONTEXT', 'FNC_COFFEE']),
  decision: z.enum(['verified', 'needs_fix', 'conditionally_approved', 'rejected', 'escalated']),
  newState: z.enum([
    'not_started', 'not_sure', 'self_serve_in_progress', 'assisted_in_progress',
    'managed_service_candidate', 'submitted', 'needs_fix', 'conditionally_approved',
    'verified', 'rejected'
  ]),
  reasonCode:   z.string().optional(),
  visibleNote:  z.string().optional(),
  internalNote: z.string().optional(),
});

const ALLOWED_TRANSITIONS: Record<string, string[]> = {
  not_started:               ['not_sure', 'self_serve_in_progress', 'assisted_in_progress'],
  not_sure:                  ['assisted_in_progress', 'managed_service_candidate'],
  self_serve_in_progress:    ['submitted', 'assisted_in_progress', 'managed_service_candidate', 'needs_fix'],
  assisted_in_progress:      ['submitted', 'managed_service_candidate', 'needs_fix'],
  managed_service_candidate: ['submitted', 'assisted_in_progress'],
  submitted:                 ['verified', 'needs_fix', 'conditionally_approved', 'rejected'],
  needs_fix:                 ['submitted', 'assisted_in_progress', 'managed_service_candidate'],
  conditionally_approved:    ['verified', 'needs_fix', 'rejected'],
  verified:                  ['needs_fix', 'rejected'],  // admin can revert verified
  rejected:                  ['not_started'],
};

router.post('/compliance/supplier/:supplierId/review', ...adminOnly, async (req, res) => {
  const sid = parseInt(req.params.supplierId, 10);
  if (isNaN(sid)) return res.status(400).json({ error: 'Invalid supplierId' });

  const parsed = reviewBodySchema.safeParse(req.body);
  if (!parsed.success) return res.status(400).json({ error: 'Invalid body', details: parsed.error.issues });
  const { requirementCode, decision, newState, reasonCode, visibleNote, internalNote } = parsed.data;

  const existing = await db.select({ state: supplierRequirementStatus.state })
    .from(supplierRequirementStatus)
    .where(and(
      eq(supplierRequirementStatus.supplierId, sid),
      eq(supplierRequirementStatus.requirementCode, requirementCode)
    ))
    .limit(1);

  if (!existing.length) {
    return res.status(404).json({ error: 'Requirement row not found. Run intake first.' });
  }

  const currentState = existing[0].state;
  const allowed = ALLOWED_TRANSITIONS[currentState] ?? [];
  if (!allowed.includes(newState)) {
    return res.status(422).json({
      error: `Transition '${currentState}' → '${newState}' not permitted`,
      allowedTransitions: allowed,
    });
  }

  await db.update(supplierRequirementStatus)
    .set({ state: newState, visibleNote, internalNote, updatedAt: new Date() })
    .where(and(
      eq(supplierRequirementStatus.supplierId, sid),
      eq(supplierRequirementStatus.requirementCode, requirementCode)
    ));

  await db.insert(adminComplianceReviews).values({
    supplierId:      sid,
    requirementCode,
    decision,
    reasonCode:      reasonCode  ?? null,
    visibleNote:     visibleNote ?? null,
    internalNote:    internalNote ?? null,
    reviewerId:      (req as any).userId,  // the reviewing officer's user ID
  });

  logger.info({ supplierId: sid, requirementCode, previousState: currentState, newState },
    'Compliance state transition');

  res.json({ ok: true, previousState: currentState, newState });
});
```

```tsx
// artifacts/fincava/src/pages/admin/ComplianceQueue.tsx
// Admin compliance priority queue — shows all suppliers ranked by compliance urgency.

import { useEffect, useState } from 'react';

interface QueueRow {
  id: number;
  display_name: string;
  sellable_status: string;
  supplier_type: string;
  ai_quality_score: number;
  open_blockers: number;
  last_reviewed_at: string | null;
  priority_score: number;
}

const STATUS_COLORS: Record<string, string> = {
  NOT_READY:  'bg-red-100 text-red-700',
  ELIGIBLE:   'bg-yellow-100 text-yellow-700',
  SELLABLE:   'bg-blue-100 text-blue-700',
  PUBLISHED:  'bg-green-100 text-green-700',
};

export default function ComplianceQueue() {
  const [rows, setRows]       = useState<QueueRow[]>([]);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    fetch('/api/admin/compliance/queue?limit=100')
      .then(r => r.json())
      .then(d => { setRows(d.queue ?? []); setLoading(false); })
      .catch(() => setLoading(false));
  }, []);

  if (loading) return <div className="p-6 text-gray-500">Cargando cola de cumplimiento...</div>;

  return (
```

```tsx
          <p className="text-sm text-gray-500 mb-3">No se ha iniciado ningún requisito CC-1.</p>
          <a href={`/admin/compliance/supplier/${supplierId}`}
            className="text-xs bg-gray-900 text-white px-3 py-1.5 rounded">
            Iniciar gestión de cumplimiento
          </a>
        </div>
      ) : (
        <div className="space-y-2">
          {data!.requirements.map(req => {
            const badge = STATE_BADGES[req.state] ?? { label: req.state, color: 'bg-gray-100 text-gray-600' };
            const path  = REQUIREMENT_PATHS[req.requirementCode];
            return (
              <div key={req.requirementCode}
                className="flex items-center justify-between py-2 border-b last:border-0">
                <div>
                  <p className="text-sm font-medium text-gray-800">
                    {REQUIREMENT_NAMES[req.requirementCode] ?? req.requirementCode}
                  </p>
                  <span className={`text-xs px-2 py-0.5 rounded ${badge.color}`}>
                    {badge.label}
                  </span>
                </div>
                {path && (
                  <a href={`/admin/compliance/supplier/${supplierId}/${path}`}
                    className="text-xs text-gray-600 underline">
                    Gestionar
                  </a>
                )}
              </div>
            );
          })}

          <div className="pt-2">
            <a href={`/admin/compliance/supplier/${supplierId}`}
              className="text-xs text-gray-500 underline">
              Ver historial completo
            </a>
          </div>
        </div>
      )}
    </div>
  );
}
```

# PASTE 5 — CC-1A: AI Compliance Write-back Hook (Enhancement)

## Prerequisite
CC-1E complete. All 7 tables in Neon. Full admin pipeline verified end-to-end.

## Context
The graduation service reads `complianceDocsTable` booleans directly, but the AI scoring output (`aiOutputsTable.complianceGaps`) is never parsed back into those booleans. This means the boolean layer reflects only onboarding declarations and manual admin updates — not AI-detected gaps.

**Invariant:** AI can only **clear** booleans (set to false if gap detected). AI cannot **verify** (set to true) — only an admin/officer can verify. This prevents false auto-verification.

---

## Task — Add Write-back to Scoring Service

Open `artifacts/api-server/src/routes/suppliers.ts`. Find the `scoreSupplier` function (search for `aiOutputsTable` insert or `export_readiness_score`).

After the AI output is written to `aiOutputsTable`, add the write-back block:

```typescript
// ── CC-1A: AI Compliance Write-back ──────────────────────────────────────────
// Parse complianceGaps from AI output and clear corresponding booleans in complianceDocsTable.
// INVARIANT: AI can only set booleans to FALSE (detected gap). Admin sets to TRUE (verified).
// This is a best-effort operation — failure does NOT fail the scoring run.

async function applyComplianceWriteback(
  supplierId: number,
  complianceGaps: string | null,
): Promise<void> {
  if (!complianceGaps) return;

  const gaps = complianceGaps.toLowerCase();

  // Detect which compliance booleans the AI flagged as missing
  const clearRut          = gaps.includes('rut') || gaps.includes('dian');
  const clearIca          = gaps.includes('ica');
  const clearFitosanitario = gaps.includes('fitosanitario') || gaps.includes('fitosanitaria');
  const clearDianExportador = gaps.includes('exportador') || gaps.includes('dian exportador');

  if (!clearRut && !clearIca && !clearFitosanitario && !clearDianExportador) return;

  const updates: Record<string, boolean> = {};
  if (clearRut)           updates.rutDian          = false;
  if (clearIca)           updates.icaRegistro       = false;
  if (clearFitosanitario) updates.fitosanitarioCert = false;
  if (clearDianExportador) updates.dianExportador   = false;

  // Only update if the current value is false already, OR if the gap is confirmed.
  // To avoid overwriting admin-set true values, we use a conditional update:
  // Only clear a boolean if it is currently FALSE (i.e., not yet admin-verified).
  // This preserves admin verifications while correcting stale "true" from onboarding input.
  // WARNING: This means AI cannot un-verify what admin verified. That is intentional.
  await db.execute(sql`
    UPDATE compliance_docs SET
      ${clearRut           ? sql`rut_dian           = CASE WHEN rut_dian           = false THEN false ELSE rut_dian END,`           : sql``}
      ${clearIca           ? sql`ica_registro        = CASE WHEN ica_registro        = false THEN false ELSE ica_registro END,`        : sql``}
      ${clearFitosanitario ? sql`fitosanitario_cert  = CASE WHEN fitosanitario_cert  = false THEN false ELSE fitosanitario_cert END,`  : sql``}
      ${clearDianExportador ? sql`dian_exportador    = CASE WHEN dian_exportador    = false THEN false ELSE dian_exportador END,`    : sql``}
      last_reviewed_at = NOW()
    WHERE supplier_id = ${supplierId}
  `);

  logger.info({ supplierId, updates }, 'CC-1A: AI compliance write-back applied');
}
```

```typescript
    'CC-1A: AI compliance write-back applied');
}
```
