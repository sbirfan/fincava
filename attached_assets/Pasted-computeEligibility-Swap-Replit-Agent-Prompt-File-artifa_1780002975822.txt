STEP 3 — Replace computeEligibility()
Find the existing computeEligibility() function. Replace the entire function
body with the following. Keep the function signature identical — same name,
same parameters, same return type.
The new function checks whether all REQUIRED requirement codes for this supplier
have a supplier_requirement_status row in verified or conditionally_approved
state. If any required code is missing or in any other state, supplier is NOT_READY.
typescript// Required requirement codes for graduation eligibility — Phase I
// CC-2 (ICA_CONTEXT) and CC-3 (FNC_COFFEE) added when those flows ship
const REQUIRED_CODES = ['DIAN_RUT'] as const;

async function computeEligibility(supplierId: number): Promise<{
  eligible: boolean;
  gaps: string[];
}> {
  const rows = await db
    .select({
      requirementCode: supplierRequirementStatus.requirementCode,
      state:           supplierRequirementStatus.state,
    })
    .from(supplierRequirementStatus)
    .where(eq(supplierRequirementStatus.supplierId, supplierId));

  const stateMap = new Map(rows.map(r => [r.requirementCode, r.state]));

  const gaps: string[] = [];

  for (const code of REQUIRED_CODES) {
    const state = stateMap.get(code);
    if (state !== 'verified' && state !== 'conditionally_approved') {
      gaps.push(code);
    }
  }

  return {
    eligible: gaps.length === 0,
    gaps,
  };
}
