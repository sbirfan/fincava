-- 1. supplier_requirement_status
-- Tracks each supplier through each compliance requirement with a full state machine.
CREATE TABLE IF NOT EXISTS supplier_requirement_status (
  id SERIAL PRIMARY KEY,
  supplier_id INTEGER NOT NULL REFERENCES suppliers(id) ON DELETE CASCADE,
  requirement_code TEXT NOT NULL,         -- e.g. 'DIAN_RUT', 'ICA_CONTEXT', 'FNC_COFFEE'
  agency TEXT NOT NULL,                   -- e.g. 'DIAN', 'ICA', 'FNC'
  state TEXT NOT NULL DEFAULT 'not_started',

  --         managed_service_candidate | submitted | needs_fix | conditionally_approved
  --         verified | rejected
  selected_mode TEXT,                     -- null | self_serve | assisted | managed
  admin_required BOOLEAN NOT NULL DEFAULT false,
  confidence_score INTEGER,               -- 0-100, optional
  visible_note TEXT,                      -- shown to supplier
  internal_note TEXT,                     -- admin-only
  created_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  updated_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),

);

// GET /api/admin/compliance/queue
// Returns all suppliers ranked by priority score (admin only)
// priority_score = (0.40 * sellable_proximity) + (0.30 * ai_quality_score)
//               + (0.20 * low_blocker_score)  + (0.10 * staleness_score)
router.get('/compliance/queue', requireAuth, requireAdmin, async (req, res) => {
  // sellable_proximity: score based on graduation_status
  //   PUBLISHED=1.0, SELLABLE=0.85, ELIGIBLE=0.5, NOT_READY=0.1
  // ai_quality_score: overall_score from supplier_evaluations (0-100) / 100
  // low_blocker_score: 1 - (open_requirement_count / 10), clamped 0-1
  //   open = state IN ('not_started','not_sure','self_serve_in_progress','needs_fix','submitted')
  // staleness_score: days_since_last_review / 30, clamped 0-1

  const queue = await db.execute(sql`
    WITH blocker_counts AS (
      SELECT supplier_id,
             COUNT(*) FILTER (WHERE state IN (
               'not_started','not_sure','self_serve_in_progress',
               'assisted_in_progress','needs_fix','submitted'
             )) AS open_blockers
      FROM supplier_requirement_status
      GROUP BY supplier_id
    ),
    last_review AS (
      SELECT supplier_id,
             MAX(reviewed_at) AS last_reviewed_at
      FROM admin_compliance_reviews
      GROUP BY supplier_id
    ),
    ai_scores AS (
      SELECT DISTINCT ON (supplier_id) supplier_id, overall_score
      FROM supplier_evaluations
      ORDER BY supplier_id, created_at DESC
    ),
    proximity AS (
      SELECT id,
        CASE graduation_status
          WHEN 'PUBLISHED'  THEN 1.00
          WHEN 'SELLABLE'   THEN 0.85
          WHEN 'ELIGIBLE'   THEN 0.50
          ELSE 0.10
        END AS sellable_proximity
      FROM suppliers
    )
    SELECT
      s.id,
      s.farm_name,
      s.graduation_status,
      COALESCE(ai.overall_score, 0) AS ai_quality_score,
      COALESCE(bc.open_blockers, 0) AS open_blockers,
      lr.last_reviewed_at,
      ROUND((
        0.40 * p.sellable_proximity +
        0.30 * (COALESCE(ai.overall_score, 0) / 100.0) +
        0.20 * GREATEST(0, 1 - COALESCE(bc.open_blockers, 0) / 10.0) +
        0.10 * LEAST(1, EXTRACT(EPOCH FROM (NOW() - COALESCE(lr.last_reviewed_at, s.created_at))) / (30 * 86400))
      )::numeric, 3) AS priority_score
    FROM suppliers s
    LEFT JOIN proximity p ON p.id = s.id
    LEFT JOIN ai_scores ai ON ai.supplier_id = s.id
    LEFT JOIN blocker_counts bc ON bc.supplier_id = s.id
    LEFT JOIN last_review lr ON lr.supplier_id = s.id
    ORDER BY priority_score DESC
    LIMIT 100
  `);

  res.json({ queue: queue.rows });
});

## TASK CC-1E: Supplier Compliance Dashboard Widget

### Instructions

Add a compliance status widget to the supplier dashboard (`/dashboard` or equivalent). This widget reads only from new tables (not from `compliance_docs` booleans). It is action-oriented — shows next step, not just status.

### New Endpoint

```typescript
// GET /api/supplier/compliance/summary
// Returns compliance status across all active requirements for this supplier
router.get('/compliance/summary', requireAuth, requireSupplier, async (req, res) => {
  const supplierId = req.user.supplierId;

  const [requirements, visibility] = await Promise.all([
    db.select().from(supplierRequirementStatus)
      .where(eq(supplierRequirementStatus.supplierId, supplierId)),
    db.select().from(buyerVisibilitySignals)
      .where(and(
        eq(buyerVisibilitySignals.supplierId, supplierId),
        eq(buyerVisibilitySignals.visible, true)
      )),
  ]);

  const visibleCount = visibility.length;
  const verifiedCount = requirements.filter(r => r.state === 'verified').length;
  const needsActionCount = requirements.filter(r =>
    ['needs_fix', 'not_started', 'not_sure'].includes(r.state)
  ).length;

  res.json({ requirements, visibleCount, verifiedCount, needsActionCount });
});
```
