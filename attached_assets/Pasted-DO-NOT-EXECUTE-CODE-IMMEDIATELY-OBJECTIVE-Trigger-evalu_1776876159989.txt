const attemptEvaluate = async (attempt = 1) => {
  try {
    await evaluateSupplier(supplierId);

    logger.info(
      { supplierId, correlationId, attempt },
      'evaluateSupplier succeeded'
    );

  } catch (err) {
    // Retry ONLY for NotFoundError (ai_outputs not ready)
    if (err.name === 'NotFoundError' && attempt <= 3) {
      const delay = Math.pow(2, attempt - 1) * 1000;

      logger.debug(
        { supplierId, correlationId, attempt, delay },
        'evaluateSupplier retry scheduled'
      );

      await new Promise(r => setTimeout(r, delay));
      return attemptEvaluate(attempt + 1);
    }

    // Final failure (or non-retryable error)
    logger.warn(
      { supplierId, correlationId, err },
      'evaluateSupplier failed'
    );

    try {
      Sentry.captureException(err);
    } catch {}
  }
};

await attemptEvaluate();
