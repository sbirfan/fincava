### New Summary Endpoint

Open `artifacts/api-server/src/routes/supplierCompliance.ts` (created in CC-1D). Add this route:

```typescript
// GET /api/supplier/compliance/summary
// Returns compliance status across all active requirements for this supplier.
// Reads ONLY from new CC-1 tables — never from compliance_docs booleans.
router.get('/summary', requireAuth, requireSupplier, async (req, res) => {
  const supplierId = req.supplierId!;

  const [requirements, visibility] = await Promise.all([
    db.select().from(supplierRequirementStatus)
      .where(eq(supplierRequirementStatus.supplierId, supplierId)),
    db.select().from(buyerVisibilitySignals)
      .where(and(
        eq(buyerVisibilitySignals.supplierId, supplierId),
        eq(buyerVisibilitySignals.visible, true)
      )),
  ]);

  const verifiedCount    = requirements.filter(r => r.state === 'verified').length;
  const needsActionCount = requirements.filter(r =>
    ['needs_fix', 'not_started', 'not_sure'].includes(r.state)
  ).length;
  const visibleCount     = visibility.length;
  const totalCount       = requirements.length; // grows as new requirements are activated

  res.json({
    requirements,
    verifiedCount,
    needsActionCount,
    visibleCount,
    totalCount,
  });
});
```

Also add `buyerVisibilitySignals` to the imports at the top of supplierCompliance.ts if not already present.

### Widget Component

Create `artifacts/fincava/src/components/supplier/ComplianceStatusWidget.tsx`:

```tsx
// artifacts/fincava/src/components/supplier/ComplianceStatusWidget.tsx
// Action-oriented compliance widget. Reads only from CC-1 new tables.
// All copy is Spanish.

import { useEffect, useState } from 'react';

interface Requirement {
  requirementCode: string;
  state: string;
  visibleNote: string | null;
}

interface SummaryResponse {
  requirements:    Requirement[];
  verifiedCount:   number;
  needsActionCount: number;
  visibleCount:    number;
  totalCount:      number;
}

// Human-readable Spanish labels per state
const STATE_LABELS: Record<string, { label: string; color: string }> = {
  not_started:              { label: 'Pendiente',          color: 'bg-gray-200 text-gray-700' },
  not_sure:                 { label: 'Necesita orientación', color: 'bg-orange-100 text-orange-700' },
  self_serve_in_progress:   { label: 'En proceso',         color: 'bg-blue-100 text-blue-700' },
  assisted_in_progress:     { label: 'Con asesor',         color: 'bg-blue-100 text-blue-700' },
  managed_service_candidate:{ label: 'Servicio gestionado', color: 'bg-purple-100 text-purple-700' },
  submitted:                { label: 'Enviado',            color: 'bg-yellow-100 text-yellow-700' },
  needs_fix:                { label: 'Necesita corrección', color: 'bg-red-100 text-red-700' },
  conditionally_approved:   { label: 'Aprobado parcialmente', color: 'bg-yellow-100 text-yellow-700' },
  verified:                 { label: 'Verificado ✓',       color: 'bg-green-100 text-green-700' },
  rejected:                 { label: 'Rechazado',          color: 'bg-red-200 text-red-800' },
};

// Spanish requirement names
const REQUIREMENT_NAMES: Record<string, string> = {
  DIAN_RUT:    'Registro DIAN (RUT)',
  ICA_CONTEXT: 'Contexto ICA',
  FNC_COFFEE:  'Registro FNC (Café)',
};

// CTA per state — what action the supplier should take next
const STATE_CTA: Record<string, string | null> = {
  not_started:              'Comenzar',
  not_sure:                 'Obtener orientación',
  self_serve_in_progress:   'Continuar',
  assisted_in_progress:     null, // waiting for advisor
  managed_service_candidate: null,
  submitted:                null, // waiting for admin
  needs_fix:                'Corregir',
  conditionally_approved:   null,
  verified:                 null,
  rejected:                 'Ver detalles',
};

// Slug for routing to individual requirement flows
const REQUIREMENT_SLUGS: Record<string, string> = {
  DIAN_RUT:    'dian-rut',
  ICA_CONTEXT: 'ica-context',
  FNC_COFFEE:  'fnc-coffee',
};

export default function ComplianceStatusWidget() {
  const [data, setData]       = useState<SummaryResponse | null>(null);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    fetch('/api/supplier/compliance/summary')
      .then(r => r.json())
      .then(d => { setData(d); setLoading(false); })
      .catch(() => setLoading(false));
  }, []);

  if (loading) return (
    <div className="border rounded-lg p-4 animate-pulse bg-gray-50">
      <div className="h-4 bg-gray-200 rounded w-48 mb-3"/>
      <div className="h-3 bg-gray-100 rounded w-full"/>
    </div>
  );

  // ── Empty state — no requirements started yet ──────────────────────────
  if (!data || data.totalCount === 0) return (
    <div className="border rounded-lg p-4 bg-white">
      <h3 className="font-semibold text-gray-800 mb-1">Requisitos de exportación</h3>
      <p className="text-sm text-gray-500 mb-3">
        Completa tus requisitos para aparecer ante compradores formales.
      </p>
      <a href="/supplier/compliance/dian-rut"
        className="block bg-green-600 text-white text-center px-3 py-2 rounded text-sm font-medium">
        Comenzar con tu registro DIAN
      </a>
    </div>
  );

  const showManagedCTA = data.needsActionCount > 0;

  return (
    <div className="border rounded-lg p-4 bg-white">
      <div className="flex items-center justify-between mb-3">
        <h3 className="font-semibold text-gray-800">Requisitos de exportación</h3>
        {data.visibleCount > 0 && (
          <span className="text-xs bg-green-100 text-green-700 px-2 py-0.5 rounded">
            {data.visibleCount} visible{data.visibleCount !== 1 ? 's' : ''} para compradores
          </span>
        )}
      </div>

      {/* Progress bar */}
      <div className="mb-4">
        <div className="flex justify-between text-xs text-gray-500 mb-1">
          <span>{data.verifiedCount} de {data.totalCount} verificados</span>
          <span>{Math.round((data.verifiedCount / data.totalCount) * 100)}%</span>
        </div>
        <div className="w-full bg-gray-100 rounded-full h-2">
          <div
            className="bg-green-500 h-2 rounded-full transition-all"
            style={{ width: `${(data.verifiedCount / data.totalCount) * 100}%` }}
          />
        </div>
      </div>

      {/* Requirement list */}
      <div className="space-y-2 mb-4">
        {data.requirements.map(req => {
          const stateInfo = STATE_LABELS[req.state] ?? { label: req.state, color: 'bg-gray-100 text-gray-600' };
          const cta       = STATE_CTA[req.state];
          const slug      = REQUIREMENT_SLUGS[req.requirementCode];

          return (
            <div key={req.requirementCode}
              className="flex items-center justify-between py-2 border-b last:border-0">
              <div>
                <p className="text-sm font-medium text-gray-800">
                  {REQUIREMENT_NAMES[req.requirementCode] ?? req.requirementCode}
                </p>
                <span className={`text-xs px-2 py-0.5 rounded ${stateInfo.color}`}>
                  {stateInfo.label}
                </span>
              </div>
              {cta && slug && (
                <a href={`/supplier/compliance/${slug}`}
                  className="text-xs bg-gray-800 text-white px-2 py-1 rounded hover:bg-gray-700">
                  {cta}
                </a>
              )}
            </div>
          );
        })}
      </div>

      {/* Managed service CTA — shown only when supplier has actionable gaps */}
      {showManagedCTA && (
        <div className="border-t pt-3">
          <p className="text-xs text-gray-500">
            ¿Necesitas ayuda completa?{' '}
            <a href="/supplier/compliance/support"
              className="text-green-700 underline font-medium">
              FINCAVA puede apoyarte con el proceso
            </a>
          </p>
        </div>
      )}
    </div>
  );
}
```
